    ```typescript
    import { useEffect, useRef, useState, useCallback } from 'react';
    import { useQueryClient } from '@tanstack/react-query';
    import { supabase } from '../lib/supabase';
    import { useAuth } from './useAuth';

    export function useRealtimeArtifacts() {
      const { isBilling, isAdmin, loading } = useAuth();
      const queryClient = useQueryClient();
      const [pendingCount, setPendingCount] = useState(0);
      const channelRef = useRef<ReturnType<typeof supabase.channel> | undefined>(undefined);

      useEffect(() => {
        // D-04 defense-in-depth + Pitfall 4 loading-race: only authorized, resolved roles subscribe
        if (loading || (!isBilling && !isAdmin)) return;

        const channel = supabase
          .channel('artifacts')
          .on(
            'postgres_changes',
            { event: 'INSERT', schema: 'public', table: 'artifacts' },
            (_payload) => {
              // Count-only — _payload data NEVER enters state (D-04)
              setPendingCount((n) => n + 1);
            }
          )
          .subscribe();

        channelRef.current = channel;
        return () => {
          void channel.unsubscribe();   // zero leak (D-04 / UI-SPEC)
        };
      }, [isBilling, isAdmin, loading]);

      const clearPending = useCallback(() => {
        setPendingCount(0);
        void queryClient.invalidateQueries({ queryKey: ['artifacts'] });
      }, [queryClient]);

      const dismissPending = useCallback(() => {
        setPendingCount(0); // D-03: dismiss without refetch
      }, []);

      return { pendingCount, clearPending, dismissPending };
    }
    ```

    import { AnimatePresence, motion, useReducedMotion } from 'framer-motion';
    import { X } from 'lucide-react';

    interface NewArtifactPillProps {
      count: number;
      onLoad: () => void;
      onDismiss: () => void;
    }

    export function NewArtifactPill({ count, onLoad, onDismiss }: NewArtifactPillProps) {
      const prefersReduced = useReducedMotion();
      const label = count === 1 ? 'Load 1 new artifact' : `Load ${count} new artifacts`;
      return (
        <AnimatePresence initial={false}>
          {count > 0 && (
            <motion.div
              role="status"
              aria-live="polite"
              initial={{ opacity: 0, y: -8 }}
              animate={{ opacity: 1, y: 0 }}
              exit={{ opacity: 0, y: -8 }}
              transition={prefersReduced ? { duration: 0 } : { duration: 0.2, ease: 'easeOut' }}
              className={cn(

                'backdrop-blur-sm border shadow-md rounded-full sm:inline-flex',
                'w-full sm:w-auto',
                // Mobile: brand-red banner; desktop: glass pill (L2)
                'bg-brand-red text-white border-transparent',
                'sm:bg-white/80 sm:text-slate-700 sm:border-slate-200'
              )}
            >
              <button
                onClick={onLoad}
                className="text-sm font-medium focus-visible:outline-none focus-visible:ring-2 focus-visible:ring-brand-red/50 rounded"
              >
                {label}

<threat_model>
## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| Supabase Realtime WebSocket → browser | INSERT change events; RLS gates per-subscriber before payload delivery |
| useAuth role state → channel subscribe | Client-side gate deciding whether to open the channel at all |
| Realtime payload → React state | Only an integer count is allowed to cross; row payload is discarded |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-06-07 | Information Disclosure | Unauthorized Realtime subscription by `pending`/anon | mitigate | Defense-in-depth: client gate `if (loading || (!isBilling && !isAdmin)) return;` (no subscribe) + RLS `artifacts_select_billing_or_admin` server-side (zero payload delivered per RESEARCH.md `has_column_privilege` finding). |
| T-06-08 | Information Disclosure | Row data leaking via the INSERT payload into `pendingCount` | mitigate | The `_payload` arg is received but NEVER stored — only `setPendingCount((n) => n + 1)` runs. Unit test asserts count-only; no row fields enter state. Toast string is a count, not row data. |
| T-06-09 | Elevation of Privilege | Stale JWT keeping a demoted user subscribed | mitigate | `supabase-js` auto-propagates `TOKEN_REFRESHED` via `realtime.setAuth()` (RESEARCH.md verified); RLS re-evaluates on refresh. Hook also re-runs its effect on `[isBilling, isAdmin, loading]` change, unsubscribing if the role drops. |
| T-06-10 | Denial of Service | Subscription leak across mount/unmount cycles | mitigate | `useEffect` cleanup calls `channel.unsubscribe()`; unit test asserts exactly-once unsubscribe on unmount. |
</threat_model>
