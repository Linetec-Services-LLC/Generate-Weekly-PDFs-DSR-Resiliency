    ```typescript
    // D-08: accept MMDDYY / MM/DD/YY / ISO, normalize toward week_ending_fmt (MMDDYY).
    export function normalizeSearchTerm(raw: string): string {
      const trimmed = raw.trim();
      const isoMatch = trimmed.match(/^(\d{4})-(\d{2})-(\d{2})$/);
      if (isoMatch) {
        const [, yyyy, mm, dd] = isoMatch;
        return `${mm}${dd}${yyyy.slice(2)}`;
      }
      return trimmed.replace(/\//g, '');
    }
    // RESEARCH Pitfall 4: .or() takes RAW PostgREST syntax with NO auto-escaping.
    // Strip the chars that break the filter string before interpolation.
    export function sanitizeSearchTerm(raw: string): string {
      return raw.replace(/[,()%]/g, '').trim();
    }
    ```

    ```typescript
    import { useState, useEffect } from 'react';

    export function useDebounce<T>(value: T, delayMs: number): T {
      const [debounced, setDebounced] = useState<T>(value);
      useEffect(() => {
        const id = setTimeout(() => setDebounced(value), delayMs);
        return () => clearTimeout(id);
      }, [value, delayMs]);
      return debounced;
    }
    ```
